CREATE TABLE Booking_list
(
    id int PRIMARY KEY AUTO_INCREMENT, 
    staff_id int NOT NULL,
    CONSTRAINT staff_id FOREIGN KEY (id) REFERENCES staff(id) ON DELETE NOTHING    
    workplase_id int NOT NULL,
    CONSTRAINT workplace_id FOREIGN KEY (id) REFERENCES workplaces(id) ON DELETE NOTHING    
    start_time Date NOT NULL,
    end_time Date NOT NULL





    login VARCHAR(255) UNIQUE,
    password VARCHAR(255),
    name VARCHAR(255),
    info VARCHAR(255),
    image VARCHAR(255),
    role int NOT NULL,
    CONSTRAINT role FOREIGN KEY (id) REFERENCES roles(id) ON DELETE CASCADE    
);

CREATE TABLE Booking_list
(
    id int PRIMARY KEY AUTO_INCREMENT, 
    staff_id int NOT NULL,
    CONSTRAINT staff_idFK FOREIGN KEY (staff_id) REFERENCES staff(id),  
    workplaсe_id int NOT NULL,
    CONSTRAINT workplace_idFK FOREIGN KEY (workplaсe_id) REFERENCES workplaces(id),    
    start_time Date NOT NULL,
    end_time Date NOT NULL   
);

CREATE TABLE WorkPlaces
(
    id int PRIMARY KEY AUTO_INCREMENT, 
    number int,
    location VARCHAR(255),
    department VARCHAR(255),
    specifications VARCHAR(255),
    IsDeleted boolean,
    IsBusy boolean
)



SELECT
    staff.id,
    staff.image,
    staff.last_name,
    staff.first_name,
    staff.patronymic,
    staff.phone_number,
    staff.email_adress,
    staff.login,
    staff.password,
    workplaces.department,
    workplaces.location,	
    workplaces.number
FROM `staff` 	
    JOIN `uniqueBooking_list` ON staff.id = uniqueBooking_list.staff_id
    JOIN `workplaces` ON uniqueBooking_list.workplaсe_id = workplaces.id
ORDER BY staff.id
